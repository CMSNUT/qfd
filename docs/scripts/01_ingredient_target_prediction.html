5. 函数1：单基因正向MR + Steiger过滤（替代SMR批量初筛）

single_gene_forward_mr <- function(gene_id){
  # 提取当前基因全部cis-eQTL SNP
  gene_eqtl <- expos_df[gene == gene_id]
  if(nrow(gene_eqtl) < 10) return(NULL) # SNP过少直接跳过
  
  # 格式化为暴露数据
  exp_raw <- format_data(
    gene_eqtl,
    type = "exposure",
    snp_col = "SNP",
    beta_col = "beta",
    se_col = "se",
    effect_allele_col = "effect_allele",
    other_allele_col = "other_allele",
    eaf_col = "eaf",
    pval_col = "pval"
  )
  
  # LD聚类去连锁（EUR 1000G）
  exp_clump <- ld_clump(
    dat = exp_raw,
    clump_r2 = 0.001,
    clump_kb = 10000,
    pop = "EUR"
  )
  if(nrow(exp_clump) < 2) return(NULL)
  
  # 匹配DCM结局SNP
  out_form <- format_data(
    outcome_df,
    type = "outcome",
    snp_col = "SNP",
    beta_col = "beta",
    se_col = "se",
    effect_allele_col = "effect_allele",
    other_allele_col = "other_allele",
    eaf_col = "eaf",
    pval_col = "pval",
    ncase_col = "ncase",
    ncontrol_col = "ncontrol"
  )
  out_sub <- extract_outcome_data(snps = exp_clump$SNP, dat = out_form)
  
  # 等位基因协调，剔除回文歧义SNP
  harm <- harmonise_data(
    exposure_dat = exp_clump,
    outcome_dat  = out_sub,
    action = 2
  )
  harm <- harm[!harm$palindromic,]
  if(nrow(harm) < 2) return(NULL)
  
  # IVW主MR
  mr_res <- mr(harm, method_list = "ivw")
  # Steiger检验：排除反向因果混杂（替代HEIDI）
  steiger_res <- steiger_filtering(harm)
  
  # 整合结果输出
  res_dt <- data.table(
    gene      = gene_id,
    ivw_b     = mr_res$b,
    ivw_se    = mr_res$se,
    ivw_p     = mr_res$pval,
    steiger_p = steiger_res$steiger_p,
    snp_count = nrow(harm)
  )
  return(list(res = res_dt, harmonise_dat = harm))
}

7. 函数2：正向MR全套（IVW/加权中位数/MR-Egger+敏感性）

run_full_forward_mr <- function(gene_id){ harm_dat <- harmonise_storage[[gene_id]] mr_all <- mr(harm_dat, method_list = c(“ivw”,“weighted_median”,“mr_egger_regression”)) mr_all\(gene <- gene_id mr_all\)direction <- “Forward(eQTL→DCM)” # logOR转OR值（DCM二分类疾病） mr_all <- mr_all %>% mutate( OR = exp(b), LCI95 = exp(b - 1.96 * se), UCI95 = exp(b + 1.96 * se) ) # 敏感性指标 het <- mr_heterogeneity(harm_dat) pleio <- mr_pleiotropy_test(harm_dat) single_snp <- mr_singlesnp(harm_dat) return(list(mr_table = mr_all, harm = harm_dat, single = single_snp)) }

run_reverse_mr <- function(gene_id){ # DCM作为暴露，提取全基因组独立IV dcm_exp_form <- format_data(outcome_df, type = “exposure”) dcm_clump <- ld_clump(dcm_exp_form, clump_r2 = 0.001, clump_kb = 10000, pop = “EUR”)

# 当前基因eQTL作为结局 gene_eqtl <- expos_df[gene == gene_id] eqtl_out_form <- format_data( gene_eqtl, type = “outcome”, snp_col = “SNP”, beta_col = “beta”, se_col = “se”, effect_allele_col = “effect_allele”, other_allele_col = “other_allele”, eaf_col = “eaf”, pval_col = “pval” ) eqtl_sub <- extract_outcome_data(dcm_clump\(SNP, eqtl_out_form) harm_rev <- harmonise_data(dcm_clump, eqtl_sub, action = 2) harm_rev <- harm_rev[!harm_rev\)palindromic,] if(nrow(harm_rev) < 2) return(NULL)

mr_rev <- mr(harm_rev, method_list = c(“ivw”,“weighted_median”,“mr_egger_regression”)) mr_rev\(gene <- gene_id mr_rev\)direction <- “Reverse(DCM→eQTL)” return(list(mr_table = mr_rev, harm = harm_rev)) }

run_single_coloc <- function(gene_id){ gene_eqtl <- expos_df[gene == gene_id] snp_inter <- intersect(gene_eqtl\(SNP, outcome_df\)SNP) eqtl_sub <- gene_eqtl[SNP %in% snp_inter] dcm_sub <- outcome_df[SNP %in% snp_inter] if(nrow(eqtl_sub) < 10) return(NULL)

# 构建coloc输入列表 dat1 <- list( beta = eqtl_sub\(beta, varbeta = eqtl_sub\)se ^ 2, type = “quant”, snp = eqtl_sub\(SNP, MAF = eqtl_sub\)eaf, N = N_eqtl ) dat2 <- list( beta = dcm_sub\(beta, varbeta = dcm_sub\)se ^ 2, type = “cc”, snp = dcm_sub\(SNP, MAF = dcm_sub\)eaf, s = s_dcm, N = N_dcm ) # 贝叶斯共定位计算 coloc_out <- coloc.abf(dataset1 = dat1, dataset2 = dat2, p1=p1, p2=p2, p12=p12) sum_dt <- as.data.table(coloc_out\(summary) sum_dt\)gene <- gene_id return(sum_dt) }
