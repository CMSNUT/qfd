source("utils.R")

compounds_dcabm <- read.xlsx("../data/raw/qfd/compounds_dcabm.xlsx") %>%
  dplyr::select(Pinyin.name, Latin.name, Blood.consitutents) %>%
  mutate(Compounds = Blood.consitutents) %>%
  # Trim leading/trailing whitespace from all character columns
  mutate(across(where(is.character), trimws)) %>%
  
  # Split multiple compounds separated by "|" into individual rows
  separate_rows(Compounds, sep = "\\|") %>%
  
  # Extract core metadata from the unstructured Compounds string
  mutate(
    CID = str_extract(Compounds, "(?<=Pubchem CID:)[^)]+") %>% str_trim(),
    DCABM.ID = str_extract(Compounds, "(?<=DCABM ID:)[^)]+") %>% str_trim(),
    temp = str_remove(Compounds, "\\s+\\(Pubchem CID:[^)]+\\)")
  ) %>%
  
  # Extract clean compound name and molecular formula
  mutate(
    temp2 = str_remove(temp, "\\([^)]+\\)$") %>% str_trim(),
    Compound = str_remove(temp2, "\\([^)]+\\)$") %>% str_trim(),
    Formula = str_match(temp2, "\\(([^)]+)\\)$")[, 2] %>% str_trim()
  ) %>%
  
  # Remove redundant intermediate columns
  dplyr::select(-Compounds, -temp, -temp2, -DCABM.ID) %>%
  
  # Final whitespace standardization for all character columns
  mutate(across(where(is.character), trimws))  %>%
  dplyr::rename(`Medicinal.materials(Latin.name)` = Latin.name,
                `Medicinal.materials(Chinese.Pinyin.name)` = Pinyin.name) %>%
  dplyr::select(-Blood.consitutents) %>%
  mutate(Source = "DCABM-TCM") %>%
  dplyr::select(
    `Medicinal.materials(Latin.name)`,
    `Medicinal.materials(Chinese.Pinyin.name)`,
    Compound,
    Formula,
    CID,
    Source
  )

1.5 Batch Retrieve PubChem Compound Annotations

compounds_pubchem <- compounds_all_processed %>%
  mutate(`CID` = clean_cids(CID)) %>%
  filter(!is.na(CID)) %>%
  distinct(CID, .keep_all = TRUE)

properties <- c("IUPACName", "MolecularFormula", "MolecularWeight",
                "SMILES")

# Split unique CIDs into batches of 20 to comply with PubChem API rate limits
unique_cids <- unique(compounds_pubchem$CID)
cid_batches <- split(unique_cids, ceiling(seq_along(unique_cids)/20))

# Set up parallel processing: use all available cores minus 2 to avoid system overload
plan(multisession, workers = availableCores() - 2)

# Execute parallel batch queries with real-time progress tracking
with_progress({
  p <- progressor(steps = length(cid_batches))
  chem_info <- future_map_dfr(cid_batches, function(b) {
    p()
    fetch_batch(properties, b)
  })
})

# Reset to sequential processing mode
plan(sequential)

# Merge PubChem annotations with the curated compound list and standardize final output
compounds_final <- compounds_all_processed %>%
  left_join(chem_info, by = "CID") %>%
  dplyr::select(
    `Medicinal.materials(Latin.name)`,
    `Medicinal.materials(Chinese.Pinyin.name)`,
    Compound, 
    IUPACName, 
    MolecularFormula, 
    MolecularWeight, 
    CID, 
    SMILES
  ) %>%
  mutate(across(where(is.character), trimws))

save(
  compounds_final,
  file = "../data/processed/02_compounds_final.RData"
)

write.xlsx(compounds_final,"../results/02_compounds_final.xlsx")

2.1 Acquisition of targets from BATMAN-TCM

rm(list = ls())

load("../data/raw/uniprot/uniprot_human_reviewed.RData") 

uniprot_human_reviewed <- uniprot_human_reviewed %>% 
  dplyr::select(Entry,Gene.Names.primary)

load("../data/processed/02_compounds_final.RData")
load("../data/raw/tcm/batman/batman_target_known.RData")
load("../data/raw/tcm/batman/batman_target_pred.RData")

# Known targets from BATMAN-TCM
compound_target_known_batman <- batman_target_known %>%
  dplyr::filter(CID %in% unique(compounds_final$CID)) %>%
  dplyr::select(CID, Uniprot.ID) %>%
  left_join(uniprot_human_reviewed, by = c("Uniprot.ID" = "Entry")) %>%
  dplyr::rename(Symbol = Gene.Names.primary) %>%
  dplyr::filter(!is.na(Symbol)) %>%
  distinct(CID, Symbol, .keep_all = TRUE) %>%
  mutate(Source = "BATMAN-Known")

# Predicted targets from BATMAN-TCM (Score ≥ 0.80,LR=48,p_adj<0.5)
compound_target_pred_batman <- batman_target_pred %>%
  dplyr::filter(CID %in% unique(compounds_final$CID)) %>%
  dplyr::select(CID, Uniprot.ID, Score) %>%
  left_join(uniprot_human_reviewed, by = c("Uniprot.ID" = "Entry")) %>%
  dplyr::rename(Symbol = Gene.Names.primary) %>%
  dplyr::filter(!is.na(Symbol), Score >= 0.80) %>%
  dplyr::select(-Score) %>%
  distinct(CID, Symbol, .keep_all = TRUE) %>%
  mutate(Source = "BATMAN-Pred")

length(unique(compound_target_pred_batman$CID))
length(unique(compound_target_pred_batman$Symbol))
length(unique(compound_target_pred_batman$Symbol)) / length(unique(compound_target_pred_batman$CID))

# Save processed BATMAN-TCM targets
save(
  compound_target_known_batman,
  compound_target_pred_batman,
  file = "../data/processed/03_batman_targets.RData"
)

source("utils.R")

load("../data/raw/uniprot/uniprot_human_reviewed.RData")
uniprot_human_reviewed <- uniprot_human_reviewed %>% 
  dplyr::select(Entry, Entry.Name, Gene.Names.primary)


# Process SEA results (confidence filters)
# P < 1e-5、MaxTc >= 0.3(modify)
compound_target_pred_sea <- read.csv("../data/raw/tcm/sea/sea-results.xls") %>%
  mutate(CID = as.numeric(gsub("compound","",Query.ID)), .before = 1) %>%
  arrange(CID) %>%
  mutate(CID = as.character(CID)) %>%
  dplyr::select(-Query.ID) %>%
  dplyr::filter(
    grepl("_HUMAN", Target.ID), 
    P.Value < 1e-10, 
    Max.Tc >= 0.3
  ) %>%
  distinct(CID, Target.ID, .keep_all = TRUE) %>%
  left_join(uniprot_human_reviewed, by = c("Target.ID" = "Entry.Name")) %>%
  dplyr::rename(Symbol = Gene.Names.primary) %>%
  dplyr::filter(!is.na(Symbol)) %>%
  dplyr::select(CID,Uniprot.ID = Entry,Symbol) %>%
  distinct(CID, Symbol, .keep_all = TRUE) %>%
  mutate(Source = "SEA")

2.5 Integrate All Targets and Deduplicate

rm(list = ls())

# Load all processed target datasets
load("../data/processed/03_batman_targets.RData")
load("../data/processed/04_swiss_targets.RData")
load("../data/processed/05_pharm_targets.RData")
load("../data/processed/06_sea_targets.RData")

# Merge all targets 
compound_target <- rbind(
  compound_target_known_batman,
  compound_target_pred_batman, 
  compound_target_pred_swiss,
  compound_target_pred_pharm,
  compound_target_pred_sea
) %>%
  group_by(CID, Symbol) %>%  
  summarise(
    Source = paste(unique(na.omit(Source)), collapse = "; "),
    across(everything(), ~ dplyr::first(.)),  
    .groups = "drop"
  )

# Filter high-confidence targets (known or predicted by ≥1  databases)
compound_target_final <- compound_target %>%
  mutate(
    is_known = str_detect(Source, "Known"),
    platform_count = str_split(Source, "; ") %>%
      map_dbl(~ length(unique(.)))
  ) %>%
  filter(
    is_known | platform_count >= 1
  ) %>%
  dplyr::select(CID,Symbol,Uniprot.ID,Source)

length(unique(compound_target_final$Symbol))

length(unique(compound_target_final$CID))

length(unique(compound_target_final$Symbol)) / length(unique(compound_target_final$CID))

# Save final target list
save(
  compound_target_final,
  file = "../data/processed/07_compound_target_final.RData"
)

write.xlsx(compound_target_final, "../results/03_compound_target_final.xlsx")
