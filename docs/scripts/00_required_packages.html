2 Clean cache

# clean 
clean_install_cache <- function() {
  unlink(file.path(tempdir(), "*"), recursive = TRUE, force = TRUE)

  cache_dir <- tools::R_user_dir("", which = "cache")
  if (dir.exists(cache_dir)) {
    unlink(file.path(cache_dir, "*"), recursive = TRUE, force = TRUE)
    message("Cleaned R cache: ", cache_dir)
  }

  gc()
  message("Install cache cleaned.")
}

clean_install_cache()
